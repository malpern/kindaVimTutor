# kindaVim — command support
#
# Authoritative list of Vim commands kindaVim does and does not
# support. Consumed at runtime by `KindaVimSupportCorpus` so the
# chat, manual, and lesson views can render a visible badge when
# an unsupported command is mentioned.
#
# Format:
#   # comments start with `#`
#   `supported:` and `unsupported:` section headers
#   one command per line (token before `—`), with a free-text
#   explanation after the em-dash
#
# Guidance:
# - List canonical command tokens WITHOUT leading counts.
#   e.g. `dw` (not `3dw`) — the matcher strips counts at lookup.
# - Include both the operator and common operator+motion pairs
#   when either might be referenced (e.g. `d` AND `dw`).
# - When in doubt err on the side of "supported" for the core
#   motion set (hjkl, word motions, operators, text objects,
#   search, visual) so false-negative badges don't pepper the UI.

supported:

# Basic motion
h — move cursor left one character
j — move cursor down one line
k — move cursor up one line
l — move cursor right one character

# Word motions
w — move to start of next word
W — move to start of next WORD (whitespace-delimited)
b — move back to start of previous word
B — move back to start of previous WORD
e — move to end of current/next word
E — move to end of current/next WORD
ge — move back to end of previous word
gE — move back to end of previous WORD

# Line motions
0 — move to first column of line
^ — move to first non-blank of line
$ — move to end of line
gg — go to first line of document
G — go to last line of document

# Paragraph and sentence motions
{ — jump to previous paragraph (blank line)
} — jump to next paragraph (blank line)
( — jump to previous sentence
) — jump to next sentence

# Character search
f — find next occurrence of character on line
F — find previous occurrence of character on line
t — move just before next occurrence of character
T — move just after previous occurrence of character
; — repeat last f/F/t/T forward
, — repeat last f/F/t/T backward

# Operators
d — delete operator
c — change operator
y — yank operator
p — put after cursor (default register)
P — put before cursor (default register)
x — delete character under cursor
X — delete character before cursor

# Operator + motion combinations
dw — delete next word
dW — delete next WORD
de — delete to end of word
dE — delete to end of WORD
db — delete to start of previous word
dB — delete back a WORD
dge — delete back to end of previous word
dgE — delete back to end of previous WORD
dh — delete one character left
dj — delete current and next line (linewise)
dk — delete current and previous line
dl — delete one character right (same as x)
d$ — delete to end of line
d0 — delete to column zero
d^ — delete to first non-blank of line
dd — delete line
D — delete to end of line (shortcut for d$)
dG — delete from current line through end of document
dgg — delete from current line back to top
dt — delete up to a character (pair with {char})
df — delete through a character (pair with {char})
dT — delete back up to a character (pair with {char})
dF — delete back through a character (pair with {char})
cw — change next word
cW — change next WORD
cb — change back a word
cB — change back a WORD
ce — change to end of word (same quirk as cw — keeps trailing space)
cE — change to end of WORD
cl — change one character (same as s)
cc — change line
c$ — change to end of line
c0 — change back to column zero
C — change to end of line (shortcut for c$)
ct — change up to a character (pair with {char})
cf — change through a character (pair with {char})
cT — change back up to a character
cF — change back through a character
yw — yank word
yW — yank WORD
yb — yank back a word
ye — yank to end of word
y$ — yank to end of line
y0 — yank back to column zero
y^ — yank to first non-blank
yt — yank up to a character (pair with {char})
yf — yank through a character (pair with {char})
yy — yank line
Y — yank line (shortcut for yy)
~ — toggle case of character under cursor

# Text objects
# kindaVim uses the `b`/`B` text-object aliases rather than raw
# paren/brace forms — `dib` / `dab` operate on `()`, `diB` / `daB`
# operate on `{}`. The raw `di(` / `di{` forms are NOT verified in
# the open-source test suite; prefer the `b` / `B` shapes here.
diw — delete inside word
daw — delete around word (includes surrounding space)
diW — delete inside WORD
daW — delete around WORD
ciw — change inside word
caw — change around word
ciW — change inside WORD
caW — change around WORD
yiw — yank inside word
yaw — yank around word
yiW — yank inside WORD
yaW — yank around WORD
dib — delete inside block (parens — standard b alias)
dab — delete around block (parens)
cib — change inside block (parens)
cab — change around block (parens)
yib — yank inside block (parens)
yab — yank around block (parens)
diB — delete inside braces (standard B alias)
daB — delete around braces
ciB — change inside braces
caB — change around braces
yiB — yank inside braces
yaB — yank around braces
di" — delete inside double-quoted string
da" — delete around double-quoted string (includes the quotes)
ci" — change inside double-quoted string
ca" — change around double-quoted string
yi" — yank inside double-quoted string
ya" — yank around double-quoted string
di' — delete inside single-quoted string
da' — delete around single-quoted string
ci' — change inside single-quoted string
ca' — change around single-quoted string
di[ — delete inside brackets
da[ — delete around brackets
ci[ — change inside brackets
ca[ — change around brackets
yi[ — yank inside brackets
ya[ — yank around brackets
ip — inside paragraph (text object)
ap — around paragraph
dip — delete inside paragraph
dap — delete around paragraph

# Visual mode
v — enter visual mode (character-wise)
V — enter visual line mode
U — uppercase selection (in visual)
viw — enter visual + inner-word text object
viW — enter visual + inner-WORD text object
vaw — enter visual + around-word text object
vaW — enter visual + around-WORD text object
vi" — enter visual + inside double quotes
va" — enter visual + around double quotes
vi' — enter visual + inside single quotes
va' — enter visual + around single quotes
vib — enter visual + inside parens block (b alias)
vab — enter visual + around parens block
viB — enter visual + inside braces block (B alias)
vaB — enter visual + around braces block
vi[ — enter visual + inside brackets
va[ — enter visual + around brackets
vip — enter visual + inside paragraph
vap — enter visual + around paragraph
viwU — select inner word then uppercase (visual-mode case operator fallback)
viw~ — select inner word then toggle case
V3j — select three lines linewise

# Search
/ — forward incremental search
? — backward incremental search
n — repeat last search forward
N — repeat last search backward

# System shortcuts (macOS — referenced for context; not Vim commands)
Cmd+S — macOS save (host app)
Cmd+Q — macOS quit (host app)
Cmd+O — macOS open (host app)
Cmd+C — macOS copy
Cmd+V — macOS paste
Cmd+T — macOS new tab (host app)
Cmd+W — macOS close window/tab (host app)
Cmd+N — macOS new window (host app)
Cmd+` — macOS cycle windows in same app
Cmd+Tab — macOS cycle between apps
Cmd+? — kindaVim Tutor help shortcut
Cmd+Space — Spotlight

# Misc
. — repeat last change
u — undo last change
Ctrl-R — redo last undone change
i — enter insert mode before cursor
I — enter insert mode at first non-blank of line
a — enter insert mode after cursor
A — enter insert mode at end of line
o — open new line below
O — open new line above
s — substitute character (delete + insert)
S — substitute line
r — replace single character
J — join next line onto current line
Esc — exit to normal mode
Enter — confirm / commit (used in search prompts, etc.)
| — move to specific column (prefixed with count)

unsupported:

# Ex commands — kindaVim runs over native text fields, no command line
# keywords: ex command, ex-command, command mode, command-mode, :w, :q, :s
: — ex command mode (no command line available)
:w — write file
:q — quit
:wq — write and quit
:x — write if changed and quit
:e — edit file
:s — substitute
:g — global command
:v — inverse global command
:r — read file
:! — run shell command

# Macros
# keywords: macro, macros, recording, playback
q — start/stop recording macro
@ — execute macro

# Registers beyond the default
# keywords: named register, registers, clipboard register
" — register prefix for named registers
"a — yank/put with register "a"
"0 — numbered register
"+ — system clipboard register (use Cmd+C/Cmd+V instead)
"* — selection register

# Windows / splits / tabs
# keywords: split, vsplit, splits, window split, tabpage, tabpages
Ctrl-W — window command prefix
:split — horizontal split
:vsplit — vertical split
:tabnew — new tab
:tabnext — next tab

# Marks beyond default
m — set mark
'' — jump to previous position (the two-single-quotes form)
`` — jump to previous position (exact)
'a — jump to mark "a"

# Folds
# keywords: fold, folds, folding, code folding
zf — create fold
za — toggle fold
zo — open fold
zc — close fold
zr — reduce fold level
zm — more fold level

# Substitute shortcuts
# keywords: substitute, :substitute
& — repeat last substitution
g& — repeat substitution over whole file

# Advanced jumps
# keywords: jumplist, tags file, ctags, go to definition
Ctrl-O — jump back (jumplist)
Ctrl-I — jump forward (jumplist)
gd — go to local definition
gD — go to global definition

# Word search shortcuts
# keywords: word search, search word under cursor
* — search forward for word under cursor
# — search backward for word under cursor
g* — loose word search forward
g# — loose word search backward

# Change-next-match workflow
# keywords: change next match, cgn workflow
cgn — change next search match
cgN — change previous search match

# Join without space
# keywords: gJ, join without space
gJ — join next line without inserting the separator space

# Match-bracket change
# keywords: c%, change to matching bracket
c% — change from cursor to matching bracket

# Raw paren / brace text-object forms

da( — delete around parens (raw form — use dab instead)
ci( — change inside parens (raw form — use cib instead)
ca( — change around parens (raw form — use cab instead)
yi( — yank inside parens (raw form — use yib instead)
ya( — yank around parens (raw form — use yab instead)
di) — delete inside parens (raw form — use dib instead)
da) — delete around parens (raw form — use dab instead)
di{ — delete inside braces (raw form — use diB instead)
da{ — delete around braces (raw form — use daB instead)
ci{ — change inside braces (raw form — use ciB instead)
ca{ — change around braces (raw form — use caB instead)
yi{ — yank inside braces (raw form — use yiB instead)
ya{ — yank around braces (raw form — use yaB instead)
vi( — visual inside parens (raw form — use vib instead)
va( — visual around parens (raw form — use vab instead)
vi{ — visual inside braces (raw form — use viB instead)
va{ — visual around braces (raw form — use vaB instead)

# Sentence / tag text objects
# keywords: sentence text object, tag text object

as — around sentence (text object)

at — around tag

# Case operators
# keywords: gU, gu, g~, case operator, uppercase operator, lowercase operator
gU — uppercase operator (pair with motion or text object)
gu — lowercase operator
g~ — toggle case operator
gUiw — uppercase inner word
guiw — lowercase inner word

# Invalid operator composition
# keywords: operator composition
